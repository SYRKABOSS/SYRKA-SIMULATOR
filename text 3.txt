  renderAll();

  log('ℹ️', `${state.members.length} members · $${state.contribution}/week · Mode: <span class="hi">${def.mode.toUpperCase()}</span>`, 'W0');
}

function renderMembers() {

    if (m.status === 'defaulted') {
      cardClass += ' danger';

      badge = '<span class="member-badge badge-gold" style="background:var(--gold-dim);color:var(--gold)">💰 Receiving</span>';

      cardClass += ' good';
      badge = '<span class="member-badge badge-good">⭐ Trust</span>';

      cardClass += ' warning';

      cardClass += ' good';
    }

    const trustColor = m.trustScore >= 80 ? 'var(--green)' : m.trustScore >= 50 ? 'var(--blue)' : 'var(--amber)';
    const exposure = m.received - m.contributed;

    return `
      <div class="${cardClass}">
        <div class="member-avatar">${m.emoji}</div>
        <div class="member-name">${m.name}</div>
        <div class="member-slot">Slot #${m.slot} · ${m.autopay?'🔄 Autopay':'⚠️ Manual'} · ${m.kyc?'✅ KYC':'❌ No KYC'}</div>
        <div class="trust-bar-wrap">
          <div class="trust-label"><span>Trust</span><span>${m.trustScore}/100</span></div>
          <div class="trust-bar"><div class="trust-fill" style="width:${m.trustScore}%;background:${trustColor}"></div></div>
        </div>
        <div class="member-amounts">
          <div class="amt"><div class="amt-label">Paid In</div><div class="amt-value" style="color:var(--green)">$${m.contributed}</div></div>
          <div class="amt"><div class="amt-label">Received</div><div class="amt-value" style="color:var(--gold)">$${m.received}</div></div>
          <div class="amt"><div class="amt-label">Exposure</div><div class="amt-value" style="color:${exposure>0?'var(--amber)':'var(--green)'}">${exposure>0?'+':''}$${exposure}</div></div>
        </div>
        ${badge}
      </div>
    `;
  }).join('');
  document.getElementById('members-grid').innerHTML = html;
  renderPayoutCard();

function renderPayoutCard() {
  const wrap = document.getElementById('payout-card-wrap');
  if (state.week === 0) { wrap.innerHTML = ''; return; }
  const recipient = state.members.find(m => m.slot === state.week);
  if (!recipient) { wrap.innerHTML = ''; return; }

  const totalHand = state.contribution * state.members.length;
  const { upfront, deferred, mode } = calcPayout(recipient);

  let cardClass, title, sub;
  if (recipient.status === 'defaulted') {
    cardClass = 'blocked';
    title = `🔴 Payout Frozen — ${recipient.name}`;
    sub = 'Member defaulted. Payout blocked until obligations are cleared.';
  } else if (mode === 'trust_full') {
    cardClass = 'full';
    title = `🟢 Full Hand Released — ${recipient.name} (Trust Mode)`;
    sub = 'Score ≥80 + KYC + autopay confirmed. Full amount transferred instantly.';
  } else {
    cardClass = 'partial';
    title = `🟡 Split Payout Active — ${recipient.name}`;
    sub = 'Standard mode: upfront portion now, remainder released over coming weeks.';
  }

  const weeksLeft = state.totalWeeks - state.week;
  const deferredPerWeek = deferred > 0 && weeksLeft > 0 ? Math.floor(deferred / weeksLeft) : 0;

  wrap.innerHTML = `
    <div class="payout-card ${cardClass}">
      <div class="payout-title">${title}</div>
      <div class="payout-sub">${sub}</div>
      <div class="payout-row"><span class="payout-key">Total Hand</span><span class="payout-val">$${totalHand}</span></div>
      <div class="payout-row"><span class="payout-key">Released Upfront</span><span class="payout-val" style="color:var(--green)">$${recipient.status==='defaulted'?0:upfront}</span></div>
      ${deferred > 0 && recipient.status !== 'defaulted' ? `
      <div class="payout-row"><span class="payout-key">Deferred (over ${weeksLeft} wks)</span><span class="payout-val" style="color:var(--amber)">$${deferred} (~$${deferredPerWeek}/wk)</span></div>` : ''}
      <div class="payout-row"><span class="payout-key">Contributed So Far</span><span class="payout-val">$${recipient.contributed}</span></div>
      <div class="payout-row"><span class="payout-key">Net Exposure</span><span class="payout-val" style="color:${(recipient.received-recipient.contributed)>0?'var(--amber)':'var(--green)'}">$${recipient.received-recipient.contributed}</span></div>
    </div>
  `;
}

function renderTimeline() {
  const html = Array.from({ length: state.totalWeeks }, (_, i) => i + 1).map(w => {
    const m = state.members.find(x => x.slot === w);
    const isActive = w === state.week;
    const isDone   = w < state.week;
    const pipColor = isDone
      ? (m && m.status === 'defaulted' ? 'var(--red)' : 'var(--green)')
      : isActive ? 'var(--gold)' : 'var(--border)';

    return `
      <div class="week-col ${isActive?'active-week':''} ${isDone&&!isActive?'done-week':''}">
        <div class="week-label">WK ${w}</div>
        <div class="week-emoji">${m ? m.emoji : '?'}</div>
        <div class="week-pip" style="background:${pipColor}"></div>
        <div class="week-name">${m ? m.name : ''}</div>
      </div>
    `;
  }).join('');
  document.getElementById('timeline').innerHTML = html;
}

function renderControls() {
  const wrap = document.getElementById('controls');
  const isRunnerActive = state.key === 'runner' && state.week >= 1 && !state.finished && !state.paused;
  const advanceLabel = state.week === 0
    ? '▶ Start Circle'
    : state.week >= state.totalWeeks
      ? '✅ Finished'
      : `⏭ Advance to Week ${state.week + 1}`;

  wrap.innerHTML = `
    <button class="btn btn-primary" onclick="advanceWeek()"
      ${state.week >= state.totalWeeks || state.finished || state.paused ? 'disabled' : ''}>
      ${advanceLabel}
    </button>
    <button class="btn btn-ghost" onclick="loadScenario('${state.key}')">🔄 Reset</button>
    ${isRunnerActive ? `<button class="btn btn-danger" onclick="triggerRunnerAttempt()">🏃 Marcus Tries to Ghost</button>` : ''}
  `;
}

// ============================================================
// PAYOUT CALC
// ============================================================
function calcPayout(member) {
  const totalHand = state.contribution * state.members.length;
  if (state.mode === 'trust' && member.trustScore >= 80 && member.kyc && member.autopay) {
    return { upfront: totalHand, deferred: 0, mode: 'trust_full' };
  }
  const positionRisk = member.slot / state.members.length;
  const remaining    = totalHand - member.contributed;
  const advanceRate  = 0.5 + (positionRisk * 0.3);
  const advance      = Math.floor(remaining * advanceRate);
  const upfront      = member.contributed + advance;
  const deferred     = totalHand - upfront;
  return { upfront, deferred, mode: 'standard_partial' };
}

// ============================================================
// ADVANCE WEEK
// ============================================================
function advanceWeek() {
  if (state.week >= state.totalWeeks || state.paused) return;
  state.week++;
  const w = state.week;
  const contrib = state.contribution;
  const totalHand = contrib * state.members.length;
  const recipient = state.members.find(m => m.slot === w);

  log('━', `<span class="gold">━━━ WEEK ${w} BEGINS ━━━</span>`, `W${w}`);

  let totalCollected = 0;

  // Collect from all payers
  state.members.forEach(m => {
    if (m.slot === w) return;
    if (m.status === 'defaulted') {
      log('⛔', `<span class="danger">${m.name} is defaulted</span> — payment skipped`, `W${w}`);
      return;
    }

    const shouldFail = m.willDefault &&
      (m.defaultAfterWeek <= w) &&
      m.status === 'active' &&
      (!m.failsOnce || w === m.defaultAfterWeek);

    if (shouldFail) {
      log('❌', `<span class="danger">${m.name}'s autopay FAILED</span> — $${contrib} not collected`, `W${w}`);
      m.retries = 0;
      setTimeout(() => retryPayment(m, w, recipient), 500);
    } else {
      m.contributed += contrib;
      m.trustScore = Math.min(100, m.trustScore + 5);
      totalCollected += contrib;
      log('✅', `<span class="good">${m.name} paid $${contrib}</span> — trust +5 → ${m.trustScore}`, `W${w}`);
    }
  });

  // Process recipient payout
  if (recipient && recipient.status !== 'defaulted') {
    recipient.contributed += contrib;
    totalCollected += contrib;
    const { upfront, deferred, mode } = calcPayout(recipient);

    if (mode === 'trust_full') {
      log('⭐', `<span class="gold">${recipient.name} receives FULL $${totalHand}</span> — Trust Mode, instant transfer`, `W${w}`);
      recipient.received += totalHand;
    } else {
      const actualUpfront = Math.min(upfront, totalCollected);
      log('💰', `<span class="gold">${recipient.name} receives $${actualUpfront} upfront</span> (of $${totalHand} hand)`, `W${w}`);
      if (deferred > 0) {
        const wksLeft = state.totalWeeks - w;
        log('⏳', `$${deferred} deferred for ${recipient.name} — ~$${Math.floor(deferred/Math.max(wksLeft,1))}/wk over ${wksLeft} weeks`, `W${w}`);
        recipient.deferredOwed = deferred;
      }
      recipient.received += actualUpfront;
    }
    recipient.exposure = recipient.received - recipient.contributed;

    // Runner post-receive log
    if (state.key === 'runner' && recipient.name === 'Marcus' && w === 1) {
      setTimeout(() => {
        log('⚠️', `<span class="warn">Note: Marcus's exposure is now $${recipient.received - recipient.contributed} — exit will be blocked</span>`, `W${w}`);
        renderAll();
      }, 700);
    }
  }

  // Release deferred tranches from prior slots
  if (w > 1 && totalCollected > 0) {
    state.members.forEach(m => {
      if (m.slot < w && m.deferredOwed > 0 && m.status !== 'defaulted') {
        const wksRemaining = state.totalWeeks - m.slot;
        const release = Math.floor(m.deferredOwed / Math.max(wksRemaining, 1));
        if (release > 0) {
          m.received += release;
          m.deferredOwed = Math.max(0, m.deferredOwed - release);
          log('📤', `Deferred release: <span class="good">${m.name} +$${release}</span> from this week's collection`, `W${w}`);
        }
      }
    });
  }

  // Cascade: Aaliya defaults week 3
  if (state.key === 'cascade' && w === 3) {
    const aaliya = state.members.find(m => m.name === 'Aaliya');
    if (aaliya && aaliya.status === 'active') {
      setTimeout(() => {
        aaliya.status = 'defaulted';
        aaliya.trustScore = Math.max(0, aaliya.trustScore - 20);
        log('💥', `<span class="danger">Aaliya's bank hard-declined — 3 retries failed → DEFAULTED</span>`, `W${w}`);
        log('🔒', `<span class="danger">Circle PAUSED · Manager notified · Aaliya blocked from future circles</span>`, `W${w}`);
        state.paused = true;
        renderAll();
      }, 800);
    }
  }

  if (w >= state.totalWeeks && !state.paused) {
    state.finished = true;
    log('🎉', `<span class="good">━━━ CIRCLE COMPLETE — all hands paid ━━━</span>`, `W${w}`);
  }

  renderAll();
}

// ============================================================
// RETRY LOGIC
// ============================================================
function retryPayment(member, week, recipient) {
  member.retries++;
  const willRecover = member.failsOnce;

  log('🔁', `Retry ${member.retries}/3 for <span class="warn">${member.name}</span> (Day ${member.retries})...`, `W${week}`);

  if (willRecover && member.retries === 2) {
    member.contributed += state.contribution;
    member.trustScore = Math.max(0, member.trustScore - 5);
    log('✅', `<span class="good">${member.name} recovered on retry 2</span> — trust −5 → ${member.trustScore}`, `W${week}`);
    renderAll();
    return;
  }

  if (member.retries < 3) {
    setTimeout(() => retryPayment(member, week, recipient), 600);
  } else {
    member.status = 'defaulted';
    member.trustScore = Math.max(0, member.trustScore - 20);
    log('⛔', `<span class="danger">${member.name} — 3 retries exhausted → DEFAULTED</span>`, `W${week}`);
    log('📩', `Manager notified · ${member.name} blocked from new circles · Trust −20 → ${member.trustScore}`, `W${week}`);
    if (recipient && recipient.deferredOwed > 0) {
      log('⏳', `<span class="warn">Deferred tranche for ${recipient.name} adjusted — ${member.name}'s share not collected</span>`, `W${week}`);
    }
    renderAll();
  }
}

// ============================================================
// RUNNER GHOST ATTEMPT
// ============================================================
function triggerRunnerAttempt() {
  const marcus = state.members.find(m => m.name === 'Marcus');
  if (!marcus) return;
  const exposure = marcus.received - marcus.contributed;

  log('🏃', `<span class="danger">Marcus attempts to disable autopay and exit the circle</span>`, `W${state.week}`);

  if (exposure > 0) {
    log('🔒', `EXIT BLOCKED — exposure: $${exposure} (received > contributed)`, `W${state.week}`);
    log('⚠️', `<span class="warn">Marcus cannot leave while he owes the circle $${exposure}</span>`, `W${state.week}`);
    log('📵', `Stripe autopay re-asserted · Account flagged · Trust −30`, `W${state.week}`);
    marcus.trustScore = Math.max(0, marcus.trustScore - 30);
    marcus.status = 'defaulted';
    renderAll();
  } else {
    log('✅', `No exposure — clean exit permitted`, `W${state.week}`);
  }
}

      <span class="log-time">${e.t}</span>
      <span class="log-icon">${e.icon}</span>
      <span class="log-text">${e.text}</span>
    </div>
  `).join('');
}
